代码生成器扩展包
composer require laracasts/generators --dev

创建表
php artisan make:migration:schema create_users_table --schema="username:string, email:string:unique"
php artisan make:migration:schema create_posts_table --schema="title:string, content:text"
添加一列
php artisan make:migration:schema add_user_id_from_posts_table --schema="user_id:integer"
删除一列
php artisan make:migration:schema remove_user_id_from_posts_table --schema="user_id:integer"
生成实体类
php artisan make:model Post --no-migration

// 重置数据库，包括重建表，和重新插入测试数据（如果有）
php artisan migrate:refresh --seed

// 更新修改到数据库
php artisan migrate

// 生成填充类
php artisan make:seeder PostsTableSeeder
// DatabaseSeeder修改
$this->call(PostsTableSeeder::class);
// 填充数据到数据库
php artisan db:seed

sudo mv /etc/apache2/envvars /etc/apache2/envvars.bak
sudo apt-get remove libapache2-mod-php5 -y
sudo apt-get install libapache2-mod-php7.1 -y
sudo cp /etc/apache2/envvars.bak /etc/apache2/envvars

sudo a2dismod php5
sudo a2dismod php5.6
sudo a2enmod php7.2
sudo service apache2 restart
